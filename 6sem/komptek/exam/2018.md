**parser.y**

```c
%{
  #include <stdio.h>
  #include <stdlib.h>
  extern int yylex ( void );
  extern char *yytext;
  int yyerror ( const char *error ) { fprintf ( stderr, "%s\n", error ); }
%}
%token NUMBER
%%
program: expr_list;
expr_list: expr | expr expr_list;
expr:
    expr expr '+' {
      // pop the two topmost elements
      printf("popq %%rax\n");
      printf("popq %%r10\n");
      // add them together and store in rax
      printf("addq %%r10, %%rax\n");
      // push rax to stack
      printf("pushq %%rax\n");
    }
  | expr expr '-' {
      // pop the two topmost elements
      printf("popq %%rax\n");
      printf("popq %%r10\n");
      // subtract them from each other and store in rax
      printf("subq %%r10, %%rax\n");
      // push rax to stack
      printf("pushq %%rax\n");
    }
  | expr expr '*' {
      // pop the two topmost elements
      printf("popq %%rax\n");
      printf("popq %%r10\n");
      // now we have to sign extend rax to rax:rdx
      printf("cqo\n");
      // now we can multiply and hope it doesnt overflow into rdx
      printf("imulq %%r10\n");
      // push rax to stack
      printf("pushq %%rax\n");
    }
  | expr expr '/' {
      // pop the two topmost elements
      printf("popq %%rax\n");
      printf("popq %%r10\n");
      // now we have to sign extend rax to rax:rdx
      printf("cqo\n");
      // now we can divide. Assuming integer division, we take the divisor that lies in rax
      printf("idivq %%r10\n");
      // push rax to stack
      printf("pushq %%rax\n");
    }
  | expr expr '%' {
      // pop the two topmost elements
      printf("popq %%rax\n");
      printf("popq %%r10\n"),
      // modulo is a case of division, first sign extend
      printf("cqo\n");
      // now divide rax:rdi by r10
      printf("idivq %%r10\n");
      // we are now interested in the remainder, which lies in rdx
      // push rdi to stack
      printf("pushq %%rdx\n");
    }
  | NUMBER {
      // put on stack the constant value
      printf("pushq $%s\n", $1);
    }
%%
#define PROGRAM_HEAD            \
".globl main\n"                 \
".section .data\n"              \
"outstr: .string \"%%ld\\n\"\n" \
".section .text\n"              \
"main:\n"                       \
"pushq %%rbp\n"                 \
"movq %%rsp,%%rbp\n"

#define PROGRAM_TAIL            \
"movq $outstr, %%rdi\n"         \
"popq %%rsi\n"                  \
"call printf\n"                 \
"leave\n"                       \
"ret\n"

int main () {
  printf ( PROGRAM_HEAD );
  yyparse();
  printf ( PROGRAM_TAIL );
}
```
